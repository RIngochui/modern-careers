```typescript
/**
 * Handle landing on APARTMENT (tile 6) or HOUSE (tile 25).
 * - Unowned: set turnPhase = WAITING_FOR_PROPERTY_DECISION, emit property-buy-prompt to active player
 * - Owned by visitor (self-land): no charge, advanceTurn
 * - Owned by someone else: charge rent (25% salary for APT, 50% for HOUSE)
 *   - Can't afford: all cash to owner, sent to Prison (independent inline mechanic)
 */
function handlePropertyTile(
  room: GameRoom, roomCode: string, playerId: string,
  tileIndex: number, roll: number, fromPosition: number
): void {
  const player = room.players.get(playerId)!;
  const tileType = BOARD_TILES[tileIndex].type; // 'APARTMENT' or 'HOUSE'
  const propKey = tileType as 'APARTMENT' | 'HOUSE';
  const ownerId = room.properties[propKey];

  // Case 1: Unowned — prompt player to buy
  if (ownerId === null) {
    const cost = propKey === 'APARTMENT' ? 50000 : 100000;
    if (player.money < cost) {
      // Can't afford — skip prompt, advance turn
      advanceTurn(room, roomCode, playerId, player.name, roll, fromPosition, tileIndex, tileType);
      return;
    }
    room.turnPhase = TURN_PHASES.WAITING_FOR_PROPERTY_DECISION;
    io.to(playerId).emit('property-buy-prompt', {
      tileIndex,
      tileName: BOARD_TILES[tileIndex].name,
      cost,
      currentMoney: player.money
    });
    return;
  }

  // Case 2: Owner lands on own property — no charge
  if (ownerId === playerId) {
    advanceTurn(room, roomCode, playerId, player.name, roll, fromPosition, tileIndex, tileType);
    return;
  }

  // Case 3: Visitor lands on owned property — pay rent
  const owner = room.players.get(ownerId)!;
  const rentRate = propKey === 'APARTMENT' ? 0.25 : 0.50;
  const rentAmount = Math.floor(player.salary * rentRate);

  if (player.money < rentAmount) {
    // Can't-pay default: all cash to owner, visitor to Prison
    const cashTransferred = player.money;
    owner.money += cashTransferred;
    player.money = 0;
    // Independent prison mechanic — do NOT call handlePrisonEscape/handlePrisonBail
    player.inPrison = true;
    player.prisonTurns = 0;
    player.inHospital = false;
    player.inJapan = false;
    player.position = 10;

    io.to(roomCode).emit('property-default', {
      tileIndex,
      visitorName: player.name,
      ownerName: owner.name,
      cashTransferred
    });
    advanceTurn(room, roomCode, playerId, player.name, roll, fromPosition, 10, 'PROPERTY_DEFAULT');
    return;
  }

  // Normal rent payment
  player.money -= rentAmount;
  owner.money += rentAmount;

  io.to(roomCode).emit('property-rent-paid', {
    tileIndex,
    visitorName: player.name,
    ownerName: owner.name,
    rentAmount
  });
  advanceTurn(room, roomCode, playerId, player.name, roll, fromPosition, tileIndex, 'PROPERTY_RENT');
}
```

**B. handleBuyProperty function** (add right after handlePropertyTile):

```typescript
/**
 * Handle player's response to property buy prompt.
 * @param accept - true = buy, false = pass
 */
function handleBuyProperty(
  room: GameRoom, roomCode: string, playerId: string, accept: boolean
): void {
  const player = room.players.get(playerId)!;
  const tileIndex = player.position;
  const tileType = BOARD_TILES[tileIndex].type;
  const propKey = tileType as 'APARTMENT' | 'HOUSE';
  const cost = propKey === 'APARTMENT' ? 50000 : 100000;

  if (accept && player.money >= cost) {
    player.money -= cost;
    room.properties[propKey] = playerId;

    io.to(roomCode).emit('property-purchased', {
      tileIndex,
      ownerName: player.name,
      tileName: BOARD_TILES[tileIndex].name,
      buyerName: player.name,
      cost
    });
  }

  advanceTurn(room, roomCode, playerId, player.name, 0, tileIndex, tileIndex, accept ? 'PROPERTY_BOUGHT' : 'PROPERTY_PASSED');
}
```
