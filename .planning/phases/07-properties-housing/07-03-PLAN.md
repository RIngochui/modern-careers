**A. initHostGame IIFE additions** — Add after the Goomba Stomp handler (after the `socket.on('goomba-stomped', ...)` block, before the closing `})();` of initHostGame):

```typescript
  // ── Phase 7: Property event handlers (host screen) ─────────────────────
  socket.on('property-purchased', ({ tileIndex, ownerName, tileName, buyerName, cost }: { tileIndex: number; ownerName: string; tileName: string; buyerName: string; cost: number }) => {
    // Update host board tile label to "[OwnerName]'s [TileName]"
    const tile = document.querySelector(`.tile[data-index="${tileIndex}"]`);
    if (tile) {
      const nameEl = tile.querySelector('.tile-name');
      if (nameEl) nameEl.textContent = `${ownerName}'s ${tileName}`;
    }
    addTurnHistory(`${buyerName} bought ${tileName} for $${cost.toLocaleString()}`);
  });

  socket.on('property-rent-paid', ({ visitorName, ownerName, rentAmount }: { visitorName: string; ownerName: string; rentAmount: number }) => {
    addTurnHistory(`${visitorName} paid $${rentAmount.toLocaleString()} rent to ${ownerName}`);
  });

  socket.on('property-default', ({ visitorName, ownerName, cashTransferred }: { visitorName: string; ownerName: string; cashTransferred: number }) => {
    addTurnHistory(`${visitorName} defaulted! $${cashTransferred.toLocaleString()} → ${ownerName}, sent to Prison`);
  });
```

**B. initPlayerGame IIFE additions** — Add after the Goomba Stomp handler (after the `socket.on('goomba-stomped', ...)` block, before the closing `})();` of initPlayerGame):

```typescript
  // ── Phase 7: Property event handlers (player screen) ───────────────────
  socket.on('property-buy-prompt', ({ tileIndex, tileName, cost, currentMoney }: { tileIndex: number; tileName: string; cost: number; currentMoney: number }) => {
    const choiceDiv = document.getElementById('property-choice');
    if (choiceDiv) {
      choiceDiv.style.display = 'block';
      choiceDiv.innerHTML = `
        <p style="margin:0 0 8px;font-weight:600;">Buy ${tileName} for $${cost.toLocaleString()}?</p>
        <p style="margin:0 0 8px;font-size:0.85rem;color:#aaa;">Your cash: $${currentMoney.toLocaleString()}</p>
        <button id="prop-buy-btn" style="margin:0 4px;padding:8px 20px;background:#4ade80;color:#1a1a2e;border:none;border-radius:6px;font-weight:700;cursor:pointer;">Buy</button>
        <button id="prop-pass-btn" style="margin:0 4px;padding:8px 20px;background:#64748b;color:#fff;border:none;border-radius:6px;font-weight:700;cursor:pointer;">Pass</button>
      `;
      document.getElementById('prop-buy-btn')?.addEventListener('click', () => {
        socket.emit('buy-property', { accept: true });
        choiceDiv.style.display = 'none';
      }, { once: true });
      document.getElementById('prop-pass-btn')?.addEventListener('click', () => {
        socket.emit('buy-property', { accept: false });
        choiceDiv.style.display = 'none';
      }, { once: true });
    } else {
      // Fallback: confirm dialog
      const buy = window.confirm(`Buy ${tileName} for $${cost.toLocaleString()}? (Your cash: $${currentMoney.toLocaleString()})`);
      socket.emit('buy-property', { accept: buy });
    }
  });

  socket.on('property-purchased', ({ buyerName, tileName, cost }: { buyerName: string; tileName: string; cost: number }) => {
    if (lastRollDisplay) lastRollDisplay.textContent = `${buyerName} bought ${tileName} for $${cost.toLocaleString()}!`;
  });

  socket.on('property-rent-paid', ({ visitorName, ownerName, rentAmount }: { visitorName: string; ownerName: string; rentAmount: number }) => {
    if (lastRollDisplay) lastRollDisplay.textContent = `${visitorName} paid $${rentAmount.toLocaleString()} rent to ${ownerName}`;
    if (statMoneyEl && mySocketId) {
      // Stat grid will be updated via next gameState broadcast
    }
  });

  socket.on('property-default', ({ visitorName, ownerName, cashTransferred }: { visitorName: string; ownerName: string; cashTransferred: number }) => {
    if (lastRollDisplay) lastRollDisplay.textContent = `${visitorName} defaulted! $${cashTransferred.toLocaleString()} to ${ownerName}. Sent to Prison!`;
    showStatusBanner(`${visitorName} couldn't afford rent! Sent to Prison.`, '#dc2626');
    setTimeout(clearStatusBanner, 4000);
  });
```
